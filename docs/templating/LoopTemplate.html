LoopTemplate class

A template loop of the form <template iterate="x in list "> or <td template repeat="x in list">.

class LoopTemplate extends PlaceholderTemplate {
 final LoopIterationSetup iterSetup;

 LoopTemplate(Node reference, exp, this.iterSetup) : super(reference, exp);

 void insert() {
   stopper = watchAndInvoke(exp, (e) {
     super.remove();
     List list = e.newValue; // watchers/observers guarantee a List
     for (int i = 0; i < list.length; i++) {
       iterSetup(list, i, this);
     }
     super.insert();
   }, 'loop-binding');
 }

 void remove() {
   super.remove();
   stopper();
   stopper = null;
 }
}
