ConditionalTemplate class

A template conditional like <template instantiate="if test"> or <td template instantiate="if test">.

class ConditionalTemplate extends PlaceholderTemplate {
 bool isVisible = false;
 final ConditionalBodySetup bodySetup;

 ConditionalTemplate(Node reference, exp, this.bodySetup)
     : super(reference, exp);

 void insert() {
   stopper = watchAndInvoke(exp, (e) {
     if (!isVisible && e.newValue) {
       // only create children when they'll be visible
       bodySetup(this);
       super.insert();
       isVisible = true;
     } else if (isVisible && !e.newValue) {
       super.remove();
       isVisible = false;
     }
   }, 'conditional-binding');
 }

 void remove() {
   super.remove();
   stopper();
   stopper = null;
 }
}
